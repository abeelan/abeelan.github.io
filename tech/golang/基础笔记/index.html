package foo

// 包里面的变量名或方法名首字母大写表示共有，可以在其他包使用
// 首字母小写表示私有

package main

import (
	"fmt"
	"go-basic/foo"
)

func main() {
	fmt.Println(foo.F)
}

上面为导包的常用方法，还可以两种方法：

import (
	"fmt"
	bar "go-basic/foo"  // <--
)

func main() {
	fmt.Println(bar.F)  // <-- 通过别名引入
}
import (
	"fmt"
	. "go-basic/foo"  // <--
)

func main() {
	fmt.Println(F)  // <-- 直接使用变量
}

数组切片

package main

import "fmt"

func main() {
	a := [3]int{1, 2, 3}
	cl := a[1:]   // [2 3]
	cl1 := a[0:1] // [1] 前开后闭
	cl2 := a[:1]  // [1] 从开始取到下标 1 结束，不包括下标 1
	cl3 := a[1:]  // [2, 3]  从下标 1 取到最后

	fmt.Println(cl1, cl2, cl3)

	// 切片是原数组的一部分，修改切片后原数组也发生改变
	cl[1] = 5
	fmt.Println(cl) // [2 5]
	fmt.Println(a)  // [1 2 5]

	// 数组容量是固定的，但是切片可以添加
	cl = append(cl, 8)
	fmt.Println(cl) // [2 5 8] 切片在原有的基础上增加了 8

	// 切片在 append 后，再修改数据，就不会影响原数组了
	cl[0] = 6
	fmt.Println(a, cl) // a[1 2 5] cl[6 5 8]

    // 原数组是没有 append copy 等方法的
    // 只有在切片上才有，创建一个空切片的方法如下
    var cli1 []int
	cli2 := make([]int, 4, 6)  // 这种方式可以直接放 len cap
}
package main

import "fmt"

func main() {
	a := [3]int{1, 2, 3}
	cl := a[:]
	fmt.Println(len(cl), cap(cl)) // 3 3
	// 向切片添加数据后，到达临界值会自动扩容，避免多次申请空间
	cl = append(cl, 5)
	fmt.Println(len(cl), cap(cl)) // 4 6
	cl = append(cl, 5)
	fmt.Println(len(cl), cap(cl)) // 5 6
	cl = append(cl, 5)
	fmt.Println(len(cl), cap(cl)) // 6 6
	cl = append(cl, 5)
	fmt.Println(len(cl), cap(cl)) // 7 12
}
package main

import "fmt"

func main() {
	a := [3]int{0, 1, 2}
	cl1 := a[:]  // [0, 1, 2]
	cl2 := a[2:] // [2]

	// 把第二个数组的内容复制到第一个数组
	copy(cl1, cl2)
	fmt.Println(cl1) // [2 1 2] 第一位有数据了就自动跳过了

	// 复制到指定切片的地方
	copy(cl1[1:2], cl2)
	fmt.Println(cl1)
}

5、map声明和使用

map 是类似于字典或哈希表的键值对，key-value

package main

import "fmt"

func main() {
	// 声明方式一 var
	var m1 map[string]string
	m1 = map[string]string{}
	// 声明方式二 隐式声明
	m2 := map[string]string{}
	// 声明方式三 make
	m3 := make(map[string]string)

	fmt.Println(m1) // map[gender:male name:tom]
	fmt.Println(m2) // map[]
	fmt.Println(m3) // map[]
}

package main

import "fmt"

func main() {
	// 基础的声明和使用
	m1 := map[string]string{}

	// 使用
	m1["name"] = "tom"
	m1["gender"] = "male"
	fmt.Println(m1)  // map[gender:male name:tom]

	// go 的 map 键可以是任意的，不局限与字符串
	m2 := map[int]bool{}
	m2[1] = true
	m2[2] = false
	fmt.Println(m2) // map[1:true 2:false]

	// 通过声明值为接口，可以放各种类型的数据，混合类型
	m3 := map[int]interface{}{}
	m3[0] = "string"
	m3[1] = 18
	m3[2] = [3]int{0, 1, 2}
	m3[3] = false
	fmt.Println(m3) // map[0:string 1:18 2:[0 1 2] 3:false]
}

package main

import "fmt"

func main() {
	m := map[string]interface{}{}
	m["a"] = 1
	m["b"] = true
	m["c"] = "test"
	m["d"] = false

	for k, v := range m {
		fmt.Println(k, v)
	}
	/*
	d false
	a 1
	b true
	c test
	*/
}

闭包函数

函数返回一个函数

package main

import "fmt"

func main() {
	closureFunc()     // 没有任何输出
	closureFunc()(10) // 输出：我是闭包函数最终执行的语句，给我的值是： 10
}

func closureFunc() func(int) {
	return func(num int) {
		fmt.Println("我是闭包函数最终执行的语句，给我的值是:", num)
	}
}

7、指针和地址

指针用 * 表示，表示这个变量本身没有存储任何值，而是拿到数据的内存地址

地址：和指针搭配使用

变量指针

package main

import "fmt"

func main() {
	var a int
	a = 123
	fmt.Println(a) // 123

	var b int
	b = a             // a 的值赋值给 b
	fmt.Println(a, b) // 123 123

	b = 456           // b 的修改不会引起 a 的变动
	fmt.Println(a, b) // 123 456

	// 重点来了 使用指针修改变量
	var c *int
	c = &a          // 把 c 指针指向 a 的内存地址，通过 &a 取变量的内存地址
	*c = 456        // 修改 c 指针所指向内存地址的值（也就是 a 的值）
	fmt.Println(a)  // 456
	fmt.Println(c)  // 0xc00010e008 指针变量输出的是内存地址
	fmt.Println(*c) // 456  通过 *c 来取该地址的值

	fmt.Println(a == *c, &a == c) // true true
}

数组指针和指针数组

package main

import "fmt"

func main() {
	var arr [3]string
	arr = [3]string{"a", "b", "c"}
	fmt.Println(arr)

	// 数组指针：数组里面都是值，指整个数组的指针地址
	var arrP *[3]string
	arrP = &arr
	fmt.Println(arrP)  // &[a b c] 数组地址
	fmt.Println(*arrP) // [a b c] 数组值
}
package main

import "fmt"

func main() {
	a := "a"
	b := "b"
	c := "c"

	array := [3]string{a, b, c}
	array[1] = "m"
	fmt.Println(b) // b ;依然是 b，数组内的值不会改变原始变量的值

	// 如果指针数组的话可以通过指针修改变量
	// 指针数组：数组里面存的都是指针
	var arr [3]*string

	arr = [3]*string{&a, &b, &c}
	fmt.Println(arr) // [0xc000014270 0xc000014280 0xc000014290]

	*arr[1] = "m"
	fmt.Println(b) // m ;通过指针更改了数组内的值
}

函数指针

package main

import "fmt"

func main() {
	s := "我是初始值"
	f(s)           // 函数的值传递
	fmt.Println(s) // 我是初始值 ;未被修改

	fPoint(&s)     // 函数的地址传递

}

func f(str string) {
	str = "f 函数体内重新赋值"
}

func fPoint(strPoint *string) {
	*strPoint = "fPoint 函数体内重新赋值"
}

8、结构体struct声明和使用

结构体是一个可以存储不同数据类型的数据类型，声明完成后，可以像其他数据类型一样使用，也有指针、值等等

声明结构体

package main

import "fmt"

func main() {
	// 通过 var 关键字声明
	var tom UserStruct
	tom.Name = "tom"
	tom.Age = 18
	tom.Male = true
	tom.Hobbies = []string{"抽烟", "喝酒", "烫头"}
	fmt.Println(tom) // {tom 18 true [抽烟 喝酒 烫头]}

	// 隐式声明
	jerry := UserStruct{
		"jerry", 18, true, []string{"吃饭", "睡觉", "打豆豆"},
	}
	fmt.Println(jerry) // {jerry 18 true [吃饭 睡觉 打豆豆]}

	// 第三种写法，比较明确
	lisa := UserStruct{
		Name:    "lisa",
		Age:     18,
		Male:    false,
		Hobbies: []string{"学习"},
	}
	fmt.Println(lisa) // {lisa 18 false [学习]}

	// 第四种写法 通过 new 一个对象
	faker := new(UserStruct)
	faker.Name = "faker"
	fmt.Println(faker) // &{faker 0 false []} 是一个地址
	// 以上所有方式不写参数都会有默认值
}

type UserStruct struct {
	// 大写开头公有 首字母小写私有
	Name    string
	Age     uint
	Male    bool
	Hobbies []string
}

package main

import "fmt"

func main() {
	tom := UserStruct{
		"tom", 18, true, []string{"吃饭", "睡觉"},
	}
	UsFunc(tom)
}

type UserStruct struct {
	// 大写开头公有 首字母小写私有
	Name    string
	Age     uint
	Male    bool
	Hobbies []string
}

func UsFunc(us UserStruct) {
	fmt.Println(us) // {tom 18 true [吃饭 睡觉]}
}

声明和使用

⭐️ goroutine

在调用一个方法的前面加上 go 就是 goroutine，让方法异步执行，相当于 协程。

package main

import "fmt"

func main() {
	Run()
}

func Run() {
	fmt.Println("Running...")
}

// 输出：Running...
func main() {
	go Run()
}

// 没有任何输出

原因是主程序开始运行后，由于 Run 函数是异步执行的，并不会阻塞当前主程序。

所以主程序就直接结束，导致 Run 还没来得及输出。

package main

	"fmt"
	"time"
)

func main() {
	go Run()
	time.Sleep(1 * time.Second)  // 主程序等待一秒再结束
}

func Run() {
	fmt.Println("Running...")
}

// 输出：Running...

在等待的一秒钟，异步函数也在执行，所以输出结果，1 秒结束后主程序停止。

这次通过一个循环打印数字的方式，看看到底是什么时候执行的 Run 函数

		fmt.Println(i)

}

func Run() {
	fmt.Println("Running...")
}

/*
	25
    26
    Running...
    27
    28
*/

每次运行后，Run 函数的输出位置都是不一样的，也就是不确定异步的 Run 函数到底什么时候执行，跟当前主线程就没关系了~控制不了。

协程管理器

如果在主线程内要等待协程执行完毕，可以使用 协程管理器。

package main

import (
	"fmt"
	"sync"
)

func main() {
	wg := sync.WaitGroup{}
	wg.Add(1)   // 设置为 1 表示正在执行
	go Run(&wg) // 这里必须传地址，否则修改无效，造成死锁
	wg.Wait()   // 等待协程执行完毕
}

func Run(wg *sync.WaitGroup) {
	fmt.Println("Running...")
	wg.Done() // Done 执行完毕后设置为 -1
}

协程间通讯工具

⭐️ channel

channel 是 goroutine 之间的通讯桥梁

定义 chan：

  • 可读可取：c:=make(chan int)
  • 可读：var readChan <- chan int = c
  • 可取：var setChan chan <- int = c
  • 有缓冲：c:=make(chan int, 5)
  • 无缓冲：c:=make(chan int)

有/无缓冲区

func main() {
	c := make(chan int, 1) // 有缓冲
	c <- 1                 // 存了 1，只能存 1 个
	fmt.Println(<-c)       // 1 读取出来 1
}

func main() {
	c := make(chan int) // 无缓冲
	c <- 1  // 没有缓冲区，存不进去
	fmt.Println(<-c) // fatal error: all goroutines are asleep - deadlock!
}

为啥报错？分析下程序执行逻辑

  • c <- 1 ，由于声明的时候，没有创建缓冲区，存不进去
  • fmt.Println(<-c)到这里要读取的时候，会先去往容量里面存，然后再读
  • 这里存的代码 c <- 1 已经执行过了，没办法在主线程再返回去调用
  • 所以程序卡在这里

如果创建一个 go 协程函数，里面存数据，那么主线程读数据，没有数据，就会读协程里面的数据，正常输出。

package main

import "fmt"

func main() {
	c := make(chan int)

	go func() {
		for i := 0; i < 10; i++ {
			c <- i
		}
	}()
	
	for i := 0; i < 10; i++ {
		fmt.Println(<-c)
	}
}

以上无缓冲区，每次打印数据的时候都先阻塞，因为没有数据，协程数据存入后，再被读取出来。

依次读（无数据）、存（协程）、打印数据，所以结果正常打印 0-9；

如果是有缓冲区，则是在携程全部存完数据后，再被主线程一次性全部打印出来。

可读可写

package main

import "fmt"

func main() {
	c := make(chan int, 5)

	var readc <-chan int = c
	var writec chan<- int = c

	writec <- 1          // 只能写
	fmt.Println(<-readc) // 读了 1
	// <-readc           // 存 1 读 1，再读就报错
}

互斥锁（Mutex）

  • Lock()
  • Unlock()
package main

import (
	"fmt"
	"time"
)

func SyncClass() {
    // 等待一秒输出一次
	lockRun()
	lockRun()
	lockRun()
}

func lockRun() {
	fmt.Println("等一秒后，执行一次")
	time.Sleep(1 * time.Second)

}

func main() {
	SyncClass()
}

上面程序每打印一次，然后就会等待一秒，直到三个函数都执行完毕，代码结束。

修改程序为通过异步执行

func SyncClass() {
    // 等待一秒输出一次
	go lockRun()
	go lockRun()
	go lockRun()

    // 阻塞下主线程 可以看到异步执行的输出信息
    time.Sleep(5 * time.Second)
}

以上执行非常快速，几乎一瞬间同时输出三条打印信息，然后等待时间结束后，程序终止。

为了解决各异步执行程序之间的共享资源问题，防止出现数据竞争和其他并发问题，可以使用互斥锁。

互斥锁最常见的使用方式是对共享资源进行加锁和解锁操作。

当一个 goroutine 需要访问共享资源时，它首先尝试获取锁：

  • 如果获取到了锁，则可以进入临界区并执行相应的操作；
  • 如果没有获取到锁，则需要等待锁被释放后再次尝试获取。

当goroutine完成对共享资源的访问后，它会将锁释放，以便其他 goroutine 可以继续访问该资源。

package main

import (
	"fmt"
	"sync"
	"time"
)

func SyncClass() {
	// 创建锁 必须是通过地址操作
	lock := &sync.Mutex{}

	go lockRun(lock)
	go lockRun(lock)
	go lockRun(lock)

	time.Sleep(5 * time.Second)
}

func lockRun(lock *sync.Mutex) {
	lock.Lock() // 上锁
    
	fmt.Println("等一秒后，执行一次")
	time.Sleep(1 * time.Second)
    
	lock.Unlock() // 执行完成后，解锁
}

func main() {
	SyncClass()
}

以上程序执行，由于主线程设置了阻塞 5 秒，所以可以看到每打印一条信息，等待一秒，然后输出下一条信息。当第一个函数 lockRun 执行时：

  • 先上锁
  • 然后输出打印信息
  • 等待一秒
  • 解锁

然后才是第二个函数进入执行，重复上述操作，这就是互斥锁的作用。

读写锁（RWMutex）

读锁：允许多人同时读，读的过程中不允许写

写锁：仅能一人同时写，写的过程中不允许读

package main

import (
	"fmt"
	"sync"
	"time"
)

func SyncClass() {
	// 创建锁 必须是通过地址操作
	lock := &sync.RWMutex{}

	go lockRun(lock)
	go lockRun(lock)
	go lockRun(lock)
	go readLockRun(lock)
	go readLockRun(lock)
	go readLockRun(lock)
	go readLockRun(lock)
	go readLockRun(lock)

	time.Sleep(5 * time.Second)
}

func lockRun(lock *sync.RWMutex) {
	// 写锁 写的时候，排斥其他的写锁和读锁
	lock.Lock()

	fmt.Println("等一秒后，执行一次")
	time.Sleep(1 * time.Second)

	lock.Unlock()
}

func readLockRun(lock *sync.RWMutex) {
	// 读锁；读锁间不互斥，可以同时读；但是会排斥写锁
	lock.RLock()

	fmt.Println("我是读锁！")
	time.Sleep(1 * time.Second)

	lock.RUnlock()
}

func main() {
	SyncClass()
}

/*
    等一秒后，执行一次
    我是读锁！
    我是读锁！
    我是读锁！
    我是读锁！
    我是读锁！
    等一秒后，执行一次
    等一秒后，执行一次
*/

程序输出分析：

  • 先执行第一把写锁，写锁执行时，不会读，其他写锁不会执行；
  • 当第一把写锁完成后，读锁全部同步执行，输出 5 次，虽然代码的调用顺序在全部写锁后；
  • 后续写锁依次按个执行。

单例（Once）

sync.Once 是一个用于实现单例模式的同步工具。sync.Once 可以确保某个函数只被执行一次，无论有多少个 goroutine 并发调用该函数。

sync.Once 类型有一个名为 Do() 的方法，用于执行传入的函数。该方法会检查一个内部变量（称为once）是否已经被设置为true，如果尚未设置，则调用传入的函数并将 once 设置为 true。

package main

import (
	"fmt"
	"sync"
)

func SyncClass() {
	one := &sync.Once{}
	for i := 0; i < 10; i++ {
		one.Do(func() {
			fmt.Println(i)
		})
	}
}
func main() {
	SyncClass()
}

以上代码，输出为 0，虽然是通过 for 循环不断打印数字，但是由于 Do 函数的限制，仅执行一次。

sync.Once 是一种非常有用的同步工具，可以用于实现单例模式、初始化操作等场景，避免竞争条件和其他并发问题，并确保程序的正确性和安全性。

WaitGroup

sync.WaitGroup 是一种同步工具，用于等待一组 goroutine 执行完毕。

WaitGroup 可以跟踪一组并发操作的状态，并在这些操作全部完成后再继续执行。

WaitGroup类型有三个方法：

  • Add()：用于将计数器加 1，表示要等待的 goroutine 数量增加；
  • Done()：用于将计数器减1，表示有一个 goroutine 已经完成；
  • Wait()：阻塞当前 goroutine，直到计数器归零
package main

import (
	"fmt"
	"sync"
	"time"
)

func SyncClass() {
	wg := &sync.WaitGroup{}
	wg.Add(2)
	go func() {
		time.Sleep(3 * time.Second)
		wg.Done()
		fmt.Println("等待了 3 秒，然后输出 Done 1")
	}()
	go func() {
		time.Sleep(5 * time.Second)
		wg.Done()
		fmt.Println("继续等待了 2 秒，然后输出 Done 2")
	}()
	wg.Wait()
	fmt.Println("Finish")
}

func main() {
	SyncClass()
}

并发字典（Map）

sync.Map 是一个线程安全的、并发访问的 map 类型。与普通的 map 不同，使用 sync.Map 类型时无需加锁即可在多个 goroutine 之间安全地并发读写数据。

sync.Map 类型的方法：

  • Load()：用于获取指定键的值；
  • Store()：用于设置指定键的值；
  • Delete()：用于删除指定键的值；
  • Range()：用于遍历所有键值对并调用指定的回调函数
  • LoadOrStore()：获取键值时如果没有就自动设置
package main

import (
	"fmt"
	"sync"
)

func main() {
	m := &sync.Map{}

	// 存储键值对
	m.Store("name", "Alice")
	m.Store("age", 18)
	m.Store("gender", "female")

	// 获取键值对
	name, _ := m.Load("name")
	age, _ := m.Load("age")
	gender, _ := m.Load("gender")
	fmt.Printf("name: %s, age: %d, gender: %s\n", name, age, gender)

	// 删除键值对
	m.Delete("gender")

	// 遍历所有键值对，当 bool 为 true 时，才进行下一次循环
	m.Range(func(key, value interface{}) bool {
		fmt.Printf("%s: %v\n", key, value)
		return true
	})
}

注意，Load()、Store() 和 Delete() 方法都不会返回任何错误信息，因此不能通过返回值来判断操作是否成功。如果需要进行错误检查，则可以在回调函数中添加额外的参数或者使用其他同步机制（如互斥锁）来保证并发安全性。

使用 sync.Map 可以避免竞争条件和其他并发问题，并确保程序的正确性和安全性。

并发池（Pool）

sync.Pool 是一种用于对象池管理的同步机制。Pool 类型可以用于缓存已经分配的临时对象，以避免频繁地创建和销毁对象，从而提高程序的性能和效率。

Pool 类型有两个方法：

  • Get()：用于从对象池中获取一个对象，如果池中没有可用对象，则会调用指定的函数生成一个新的对象并返回；
  • Put()：用于将不再使用的对象放回池中，以便下次可以被重复使用。
package main

import (
	"fmt"
	"sync"
)

func main() {
	p := &sync.Pool{}

	// 放入 3 个对象
	p.Put(1)
	p.Put(2)
	p.Put(3)

    // 取出 3 个对象
	for i := 0; i < 3; i++ {
		fmt.Println(p.Get())
	}

	// 当池子里的对象取完后再取就是 nil
	fmt.Println(p.Get())
}

由于 Pool 类型的实现方式不同于普通的变量和对象，在使用时需要特别小心。例如，在对于 sync.Pool 类型的对象不能进行拷贝、比较和序列化等操作，否则可能会导致程序崩溃或出现其他异常。

Cond

sync.Cond 是一种条件变量，用于在多个 goroutine 之间实现等待、通知和唤醒操作。

与互斥锁不同，使用 Cond 类型时可以使某些 goroutine 进入等待状态，直到满足特定条件后再被唤醒。

Cond 类型有三个方法：

  • Wait()：用于使当前 goroutine 进入等待状态，并释放与 Cond 关联的互斥锁；
  • Signal()：用于唤醒一个处于等待状态的 goroutine；
  • Broadcast()：用于唤醒所有处于等待状态的 goroutine。
package main

import (
	"fmt"
	"sync"
	"time"
)

// 将 cond 和 互斥锁关联
var cond = sync.NewCond(&sync.Mutex{})

// 定义全局切片，可增删元素
var queue []int

// 模拟生产一个元素，并添加到队列
func produce() {
	for i := 0; i < 10; i++ {
		time.Sleep(time.Millisecond * 500)
		cond.L.Lock()
		queue = append(queue, i)
		fmt.Printf("produce: %d\n", i)

		// 生产完成后唤醒一个等待状态的 goroutine
		cond.Signal()
		cond.L.Unlock()
	}
}

// 模拟消费一个元素，并将其从队列中删除
func consume() {
	for i := 0; i < 10; i++ {
		cond.L.Lock()

		// 如果队列为空，则进入等待状态
		for len(queue) == 0 {
			cond.Wait()
		}

		// 被唤醒则消费一个
		item := queue[0]
		queue = queue[1:]
		fmt.Printf("consume: %d\n", item)
		cond.L.Unlock()
	}
}

func main() {
	go produce()
	consume()
}

如果消费者大于生产者，那么当队列为空时，消费者 goroutine 会一直阻塞在 cond.Wait() 方法上，导致程序出现死锁的情况。

TCP

实现一个服务端和一个客户端，进行消息通信

package main

import (
	"fmt"
	"net"
)

func main() {
	// 监听本地 8888 端口
	listenAddr, err := net.ResolveTCPAddr("tcp", ":8888")
	if err != nil {
		fmt.Println("Error resolving TCP address:", err.Error())
		return
	}

	// 创建 TCP 监听器
	listener, err := net.ListenTCP("tcp", listenAddr)
	if err != nil {
		fmt.Println("Error listening on TCP port:", err.Error())
		return
	}

	defer listener.Close()
	fmt.Println("Server is listening on ", listener.Addr().String())

	// 不断接受新的连接，并处理客户端发送的数据
	for {
		conn, err := listener.AcceptTCP()
		if err != nil {
			fmt.Println("Error accepting new connection:", err.Error())
			return
		}

		fmt.Printf("New client connected: %s\n", conn.RemoteAddr().String())
		go handleConnection(conn) // 处理客户端连接
	}
}

func handleConnection(conn *net.TCPConn) {
	defer conn.Close()

	for {

		// 读取客户端发送的数据
		buf := make([]byte, 1024)
		n, err := conn.Read(buf)
		if err != nil {
			fmt.Println("Error reading data from client:", err.Error())
			break
		}

		// 输出客户端发送的数据
		data := string(buf[:n])
		fmt.Printf("[ Client %s ]: %s\n", conn.RemoteAddr().String(), data)

		// 返回给客户端信息
		conn.Write([]byte("收到了消息: " + data))
	}
}
package main

import (
	"bufio"
	"fmt"
	"net"
	"os"
)

func main() {
	// 连接本地 8888 端口
	serverAddr, err := net.ResolveTCPAddr("tcp", "127.0.0.1:8888")
	if err != nil {
		fmt.Println("Error resolving TCP address:", err.Error())
		return
	}

	// 建立连接
	conn, err := net.DialTCP("tcp", nil, serverAddr)
	if err != nil {
		fmt.Println("error connecting to TCP port:", err.Error())
		return
	}
	defer conn.Close()
	fmt.Println("Connected to server:", conn.RemoteAddr().String())

	// 向服务器发送数据
	reader := bufio.NewReader(os.Stdin)
	for {
		bytes, _, _ := reader.ReadLine()
		_, err = conn.Write(bytes)
		if err != nil {
			fmt.Println("Error writing data to server:", err.Error())
			return
		}

		reply := make([]byte, 1024)
		buf, _ := conn.Read(reply)
		fmt.Println(string(reply[:buf]))
	}
}

HTTP

  • server：服务端，包含地址、端口、处理器等等
  • conn：用户请求来的那个链接
  • response：响应
  • request：用户请求信息
  • Handle：对于接收的信息进行处理并且返回的一个处理器
package main

import (
	"io"
	"net/http"
)

func handle(res http.ResponseWriter, req *http.Request) {
	switch req.Method {
	case "GET":
		res.Write([]byte("[Get]"))
		break
	case "POST":
		// 将请求体返回去
		b, _ := io.ReadAll(req.Body)
		res.Write(b)

		// 设置 headers
		header := res.Header()
		header["test"] = []string{"test1", "test2"}

		// 设置状态码
		res.WriteHeader(http.StatusBadRequest) // 设置返回状态码为 400
		break
	}
}

func main() {
	http.HandleFunc("/test", handle)
	http.ListenAndServe(":8080", nil)
}
